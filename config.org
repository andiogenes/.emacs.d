** Install
#+begin_src emacs-lisp
  (setq the-packages
        '(
          ace-window
          powerline
          dashboard

          clojure-mode
          cider
          ))

  (mapcar (lambda (package)
            (if (package-installed-p package)
                nil
              (package-install package)))
          the-packages)
#+end_src
